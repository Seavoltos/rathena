L_Enter:
	switch(instance_enter("Assault on the Airship")) {
	case IE_OTHER:
		mes "Un error desconocido ha ocurrido.";
		close;
	case IE_NOINSTANCE:
		mes "La instancia no existe o el lider de la Party no la ha generado.";
		close;
	case IE_NOMEMBER:
		mes "Sólo puedes ingresar luego de hacer una Party";
		close;
	case IE_OK:
		mapannounce "instance", strcharinfo(0) +" de la Party, "+ getpartyname( getcharid(1) ) +", ha entrado a la dungeon, "+.@md_name$+".",bc_map,"0x00ff99",FW_NORMAL,12;
		if (getarg(1)) {
			set aasault_timer,gettimetick(2);
			set aasault_partyid, getcharid(1);
			setquest 15050;
		}
		if (getarg(0) == 0) close;
		else end;
	}
}

		mes "Dificultad " + .dificultad$[.@i];
		if( .@i == 4 )
			close;
		else
			next;

	close;

	for (set .@i, 1; .@i <= 4; set .@i, .@i + 1) {

		setd "$120903_num_AAsault"+'120903_str_AAsaultDif , .@AAsault_member;
	}
	end;

OnTimer3600000:
	stopnpctimer;
	end;
OnInstanceInit:
	hideonnpc instance_npcname("#globaltimer51");
	end;
